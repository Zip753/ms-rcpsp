54  bool operator()(const std::pair<int, int> a,
55  const std::pair<int, int> b) const {
56  return a.second < b.second;
57  }
58  };

60  // heap of pairs (task_id, priority)
61  std::priority_queue<std::pair<int, int>,
62  std::vector<std::pair<int, int>>,
63  PriorityComp> queue;

65  if (tasks[i]->dep_size() == 0) {
66  queue.push(std::make_pair(i, prio[i]));
67  }
68  }

70 
71  // availability time for resources

82  queue.pop();
83 
84  // find max finish time of all dependencies
85  int res_idx = resource(itask);
86  int min_start = 0;
87  for (int i = 0; i < tasks[itask]->dep_size(); ++i) {
88  int idep = tasks[itask]->dep[i];
89  int fin = finish_time(idep);
90  if (min_start < fin)
91  min_start = fin;
92  }

96  // update availability time for resource
97  time[res_idx] = finish_time(itask);

102  dep_count[inext]++;
103  if (dep_count[inext] == tasks[inext]->dep_size()) {
