58 void PrioSchedule::fix_all() {
59  struct PriorityComp {
60  bool operator()(const std::pair<int, int> a,
61  const std::pair<int, int> b) const {
62  return a.second < b.second;
63  }
64  };
65 
66  // heap of pairs (task_id, priority)
67  std::priority_queue<std::pair<int, int>,
68  std::vector<std::pair<int, int>>,
69  PriorityComp> queue;
70  for (int i = 0; i < n; ++i) {
71  if (tasks[i]->dep_size() == 0) {
72  queue.push(std::make_pair(i, prio[i]));
73  }
74  }
75  int res_count = Project::get()->get_res_count();
76 
77  // availability time for resources
78  int* time = new int[res_count];
79  std::fill_n(time, res_count, 0);
80 
81  // number of complete dependencies for tasks
82  int* dep_count = new int[n];
83  std::fill_n(dep_count, n, 0);
84 
85  while (!queue.empty()) {
86  // take next task
87  int itask = queue.top().first;
88  queue.pop();
89 
90  // find max finish time of all dependencies
91  int res_idx = resource(itask);
92  int min_start = 0;
93  for (int i = 0; i < tasks[itask]->dep_size(); ++i) {
94  int idep = tasks[itask]->dep[i];
95  int fin = finish_time(idep);
96  if (min_start < fin)
97  min_start = fin;
98  }
99 
100  // update start time for the task
101  start[itask] = std::max(min_start, time[res_idx]);
102  // update availability time for resource
103  time[res_idx] = finish_time(itask);
104 
105  // add all unblocked dependent tasks to the queue
106  for (int i = 0; i < tasks[itask]->next_size; ++i) {
107  int inext = tasks[itask]->next[i];
108  dep_count[inext]++;
109  if (dep_count[inext] == tasks[inext]->dep_size()) {
110  queue.push(std::make_pair(inext, prio[inext]));
111  }
112  }
113  }
114 
115  delete[] time;
116  delete[] dep_count;
117 }
